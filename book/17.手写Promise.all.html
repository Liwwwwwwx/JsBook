17.手写Promise.all


function MyAll(prams){
    let res, rej
    var p = new Promise((resolve, reject) => {
        res = resolve
        rej = reject
    })

    let result = []
    let count = 0
    let fullFillCount = 0
    for (const pram of prams) {
        const i = count
        count++
        Promise.resolve(pram).then((data) => {
            result[i] = data
            fullFillCount ++
            if(fullFillCount == count) {
                res(result)
            }
        },rej)
    }

    if(count == 0 ) {
        res(result)
    }

    return p
}

MyAll([1,2,3,4]).then((data) => {
    console.log(data)
})

Promise.all([1,2,3,4]).then((data)=> {
    console.log(data)
})
